3 Example PHP Code

Here's a sample PHP code you can try:

<?php
class UserManager {
    private $pdo;
    
    public function __construct($database) {
        $this->pdo = new PDO($database);
    }
    
    public function createUser($username, $email) {
        $stmt = $this->pdo->prepare("INSERT INTO users (username, email) VALUES (?, ?)");
        return $stmt->execute([$username, $email]);
    }
    
    public function getUser($id) {
        $stmt = $this->pdo->prepare("SELECT * FROM users WHERE id = ?");
        $stmt->execute([$id]);
        return $stmt->fetch(PDO::FETCH_ASSOC);
    }
}

$userManager = new UserManager("mysql:host=localhost;dbname=mydb");
$user = $userManager->getUser(1);
echo json_encode($user);
?>

import sqlite3
from typing import Optional, Dict, Any

class UserManager:
    def __init__(self, database: str):
        self.connection = sqlite3.connect(database)
        self.connection.row_factory = sqlite3.Row
    
    def create_user(self, username: str, email: str) -> bool:
        try:
            cursor = self.connection.cursor()
            cursor.execute("INSERT INTO users (username, email) VALUES (?, ?)", 
                         (username, email))
            self.connection.commit()
            return True
        except Exception:
            return False
    
    def get_user(self, user_id: int) -> Optional[Dict[str, Any]]:
        cursor = self.connection.cursor()
        cursor.execute("SELECT * FROM users WHERE id = ?", (user_id,))
        result = cursor.fetchone()
        return dict(result) if result else None

# Usage
user_manager = UserManager("mydb.sqlite")
user = user_manager.get_user(1)
print(json.dumps(user) if user else "User not found")
